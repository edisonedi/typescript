/**
 * Todo 앱을 만들기 위한 클래스
 * @constructor
 */
class TodoListApp {
  constructor() {}

  /**
   * 할 일을 추가할 수 있다.
   * 내용없이 추가할 수 없다.
   * @param {CreateTodo} createTodo
   */
  createTodo({ content, isCompleted = false, category = "", tags = [] }) {}

  /**
   * 모든 할 일을 조회할 수 있다.
   * @return {Todo[]} todoList
   */
  readAllTodoList() {}

  /**
   * ID를 기반으로 특정 할 일을 조회할 수 있다.
   * @param {id} id
   * @return {Todo} todo
   */
  readTodoById(id) {}

  /**
   * ID를 제외한 모든 속성을 수정할 수 있다.
   * 특정 할 일의 특정 태그를 수정할 수 있다.
   * @param {id} id
   * @param {UpdateTodo} updateTodo
   */
  updateTodoById(id, { content, isCompleted, category, oldTag, newTag }) {}

  /**
   * ID를 기반으로 특정 할 일을 삭제할 수 있다.
   * @param {id} id
   */
  deleteTodoById(id) {}

  /**
   * 특정 할 일의 특정 태그를 삭제할 수 있다.
   * @param {id} id
   * @param {tag} tag
   */
  deleteTagById(id, tag) {}

  /**
   * 특정 할 일의 특정 태그를 삭제할 수 있다.
   * @param {id} id
   */
  deleteTagsById(id) {}

  /**
   * 모든 할 일을 제거할 수 있다.
   */
  deleteAllTodoList() {}
}

const todoApp = new TodoListApp();

todoApp.createTodo({ id: 1, content: "" });
